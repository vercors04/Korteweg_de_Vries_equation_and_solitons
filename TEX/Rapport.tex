\begin{frame}{Introduction}
\textbf{Contexte :} Observation 1834 d'une vague solitaire se propageant sans se dissiper. 
Formalisation en 1895 par Korteweg et de Vries. 
\[
\frac{\partial u}{\partial t} + \varepsilon u \frac{\partial u}{\partial x} + \mu \frac{\partial^3 u}{\partial x^3} = 0
\]
$\varepsilon = 0{,}2$, $\mu = 0{,}1$.
\bigskip

\textbf{Déroulement :}
\begin{enumerate}
    \item Solution analytique
    \item Étude du schéma numérique
    \item Premiers résultats
    \item Lien amplitude -- vitesse
    \item Collision de deux solitons
    \item Décomposition en solitons
\end{enumerate}
\end{frame}

\begin{frame}{Solution analytique}
\[
\boxed{u(x,t) = \frac{3c}{\varepsilon} \text{sech}^2\left[ \frac{1}{2}\sqrt{\frac{c}{\mu}}(\xi - \xi_0) \right]}
\]

avec $\xi=x-ct$, $\xi_0$ position initiale et $\text{sech}=\frac{1}{\cosh{x}}$

\begin{frame}{Schémas numériques}

\hspace{0.5cm}\textbf{Terme non linéaire :} $\varepsilon u \,\dfrac{\partial u}{\partial x}$

\medskip

\textbf{Leap-frog} — explicite, conditionnellement stable - \textit{$\mathcal{O}(h_t),\ \mathcal{O}(h_x^2)$}\\
$\rightarrow$ Condition CFL $h_t \leq C \cdot h_x^3$ — condition stricte

\medskip

\textbf{Crank-Nicolson} — semi-implicite, inconditionnellement stable
\hspace{2cm}\textit{$\mathcal{O}(h_t),\ \mathcal{O}(h_x^2)$}\\
$\rightarrow$ Terme non linéaire figé : $\varepsilon u_i^{[n]} \cdot \partial_x u^{[n+1/2]}$

\medskip

\textbf{CN Prédicteur-Correcteur} — semi-implicite, inconditionnellement stable \hspace{2cm} \textit{$\mathcal{O}(h_t^2),\ \mathcal{O}(h_x^2)$}\\
$\rightarrow$ Prédiction $U^*$ par CN classique, puis $u_{\text{milieu}} = \dfrac{u^{[n]}+u^*}{2} \approx u(t_{n+1/2})$

\end{frame}
\begin{frame}{Comparaison des schémas}
Domaine $x \in [-15,15]$, $N_x=1000$, $N_t=700$, $T=10$.

CN : perte d'amplitude du à la linéarisation (précision en $\mathcal{O}(h_t)$)

CN-PC : Erreur largement plus faible, produit de bons résultats

Leap-Frog : Nécessite une réduction de la précision spatiale importante pour satisfaire la CFL. 
\end{frame}

\begin{frame}{Conclusion}

\textbf{Discrétisation}

\medskip
\hspace{1em} Le terme non linéaire est un enjeu — sa gestion au temps $t_{n+1/2}$\\
\hspace{1em} via le schéma prédicteur-correcteur restaure la précision d'ordre 2

\bigskip\bigskip

\textbf{Caractéristiques des solitons}

\medskip
\hspace{1em} La vitesse dépend de l'amplitude\\
\hspace{1em} Collision : amplitude, vitesse et forme inchangées

\bigskip\bigskip

\textbf{Conditions initiales}

\medskip
\hspace{1em} Une simple discontinuité engendre une hiérarchie de solitons,\\
\hspace{1em} illustration des comportements précédemment évoqués
